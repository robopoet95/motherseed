73       public ArrayList<RegisterActivity> readDatabase() {
74           SQLiteDatabase database = this.getReadableDatabase();
75   
76           Cursor cursor = database.rawQuery("SELECT * FROM " + TABLE_NAME, null);
77   
78           ArrayList<RegisterActivity> registerActivityArrayList = new ArrayList<>();
79   
80           if (cursor.moveToFirst()) {
81               do {
82                   RegisterActivity registerActivity = new RegisterActivity();
83                   registerActivity.setUsername(cursor.getString(cursor.getColumnIndex(COL_USERNAME)));
84                   registerActivity.setPassword(cursor.getString(cursor.getColumnIndex(COL_PASSWORD)));
85                   registerActivity.setFirstName(cursor.getString(cursor.getColumnIndex(COL_FIRST_NAME)));
86                   registerActivity.setSecondName(cursor.getString(cursor.getColumnIndex(COL_SECOND_NAME)));
87                   registerActivity.setEmailId(cursor.getString(cursor.getColumnIndex(COL_EMAIL_ID)));
88                   registerActivity.setDateOfBirth(cursor.getString(cursor.getColumnIndex(COL_DOB)));
89                   registerActivity.setAddress(cursor.getString(cursor.getColumnIndex(COL_ADDRESS)));
90                   registerActivity.setPhNumber(cursor.getString(cursor.getColumnIndex(COL_PH_NUMBER)));
91   
92                   registerActivityArrayList.add(registerActivity);
93               } while (cursor.moveToNext());
94           }
95   
96           cursor.close();
97           database.close();
98   
99           return registerActivityArrayList;
100      }

102      public void updateUser(RegisterActivity registerActivity) {
103          SQLiteDatabase database = this.getWritableDatabase();
104  
105          ContentValues values = new ContentValues();
106  
107          values.put(COL_USERNAME, registerActivity.username);
108          values.put(COL_PASSWORD, registerActivity.password);
109          values.put(COL_FIRST_NAME, registerActivity.firstName);
110          values.put(COL_SECOND_NAME, registerActivity.secondName);
111          values.put(COL_EMAIL_ID, registerActivity.emailId);
112          values.put(COL_DOB, registerActivity.dateOfBirth);
113          values.put(COL_ADDRESS, registerActivity.address);
114          values.put(COL_PH_NUMBER, registerActivity.phNumber);
115  
116          database.update(TABLE_NAME, values, COL_EMAIL_ID + " = ?", new String[] {String.valueOf(registerActivity.getEmailId())});
117          database.close();
118      }

139          int cursorCount = cursor.getCount();

142          database.close();
